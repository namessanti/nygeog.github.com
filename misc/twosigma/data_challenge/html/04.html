import pandas as pd
import matplotlib.pyplot as plt
import datetime
import numpy as np
from _settings import * 

df = pd.read_csv(d+'processing/annual_sales/all.csv')
print len(df.index) #Counts NaN values
df = df[(df.sale_year < 2015)]  #all sales less than 2015
print len(df.index), ' - rows after removing 2015 and above'
df = df[(df.sale_price > 0)]  #all sales less than 2015
print len(df.index), ' - rows after only keeping sale_price > 0'

df = df[(df.neighborhood == 'BROOKLYN HEIGHTS         ') | (df.neighborhood == 'BEDFORD STUYVESANT       ')]

df['price_tier'] = np.where(df['sale_price'] < 3000000, '< 3mil', '>= 3mil')

df = df[['price_tier','sale_year','sale_price','neighborhood']]

df['count'] = 1

priceStats = df.sale_price.describe()

dfg = df.groupby(['price_tier','sale_year','neighborhood'],as_index=False).sum()#,as_index=False).sum()
dfg['index_col'] = dfg.index
dfg = dfg.sort(['sale_year','neighborhood','price_tier'],ascending=[True,True,True])

#print dfg.head(25)

df = dfg

Series = (df.sale_price - df.groupby(['price_tier','neighborhood']).sale_price.shift(1))/ df.sale_price

df = Series.to_frame(name='yoy')
df['index_col'] = df.index
df = df.merge(dfg, how='left', on='index_col')

df = df[['price_tier','yoy','sale_year','neighborhood']]

#print df.head(100)

df1 = df[(df.price_tier == '< 3mil' ) & (df.neighborhood == 'BROOKLYN HEIGHTS         ')]
df2 = df[(df.price_tier == '>= 3mil') & (df.neighborhood == 'BROOKLYN HEIGHTS         ')]
df3 = df[(df.price_tier == '< 3mil' ) & (df.neighborhood == 'BEDFORD STUYVESANT       ')]
df4 = df[(df.price_tier == '>= 3mil') & (df.neighborhood == 'BEDFORD STUYVESANT       ')]

plt.plot(df1['sale_year'],df1['yoy'],color='orange'  ,label='$Brooklyn \ Heights < \$ 3mil$')
plt.plot(df2['sale_year'],df2['yoy'],color='darkred' ,label='$Brooklyn \ Heights  \geq \$ 3mil$')
plt.plot(df3['sale_year'],df3['yoy'],color='green'   ,label='$Bedford \ Stuyvesant < \$ 3mil$')
plt.plot(df4['sale_year'],df4['yoy'],color='darkblue',label='$Bedford \ Stuyvesant \geq \$ 3mil$')

plt.legend(loc=3)
plt.title("2004-2014 YoY Sales split by Sale Price above/below $3mil (2003-2014 sales)")

plt.legend(loc = 'center left', bbox_to_anchor = (1.0, 0.5))

plt.show()
