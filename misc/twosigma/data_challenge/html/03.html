import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from matplotlib import rcParams
rcParams.update({'figure.autolayout': True}) #http://stackoverflow.com/questions/6774086/why-is-my-xlabel-cut-off-in-my-matplotlib-plot
from _settings import *

df = pd.read_csv(d+'processing/annual_sales/res.csv')
df = df[(df.sale_year < 2015)]  #all sales less than 2015

df = df[['year_built','sale_year']]

df['dif_built_sale'] = df['sale_year'] - df['year_built'] #if sale_year is less than year_built, its still a valid new build, getting some neg. numbers

df['new_construction'] = np.where(df['dif_built_sale'] > 2,'old build', 'new build') 

df = df.sort('dif_built_sale',ascending=False)

#print df.head(20)
#print df.tail(20)

df['count'] = 1
df = df[['new_construction','sale_year','count']]

df['year_newold'] = df.sale_year.map(str) + ' - ' + df['new_construction']
df = df[['year_newold','count']]

df = df.groupby(['year_newold'],as_index=False).sum()#,as_index=False).sum()
df = df.sort(['year_newold'],ascending=[True])
df = df.set_index('year_newold')

#print df.head(10)

ax = df['count'].plot(kind='bar', title ="Count of Sales by New Construction (building built year more than 2 years from sale year)",figsize=(15,10),legend=True, fontsize=12)
ax.set_xlabel("New Build vs. Old Build (Greater than 2 years old) over Years",fontsize=12)
ax.set_ylabel("Count",fontsize=12)
plt.show()
