df = pd.read_csv(d+'processing/annual_sales/res.csv')

df = df[['neighborhood','sale_year','sale_price']]

df = df[(df.sale_year == 2014) | (df.sale_year == 2013)]
df['count'] = 1

dfg = df.groupby(['neighborhood','sale_year'],as_index=False).sum()#,as_index=False).sum()
dfg['index_col'] = dfg.index
#print dfg.head(25)

df = dfg

#df = df.set_index('neighborhood')
Series = (df.sale_price - df.groupby(['neighborhood']).sale_price.shift(1))/ df.sale_price

df = Series.to_frame(name='yoy')
df['index_col'] = df.index
df = df.merge(dfg, how='left', on='index_col')

df = df.sort(['yoy'], ascending=False)

#print df.head(10)
df = df[(df.sale_year == 2014)]

#print df.dtypes
#print df.head(10)

df['weird'] = df.neighborhood.str.contains('UNKNOWN') #flagged if neighborhood has string 'UNKNOWN'
df = df[(df.weird == False)] #remove all flagged
min2014Sales = 100
df = df[(df['count'] >= min2014Sales )] #Lets say for YoY you need 100 sales at least in 2014. 

df = df[['neighborhood','yoy']]
df = df.sort(['yoy'], ascending=False)

df = df.head(30)

#print df.head(100)
df = df.set_index('neighborhood')

ax = df['yoy'].plot(kind='bar', title ="Top 30 Neighborhoods Year over Year Residential Sales (2013-2014) with at least "+str(min2014Sales)+" sales",figsize=(15,10),legend=True, fontsize=12)
ax.set_xlabel("Neighborhood",fontsize=12)
ax.set_ylabel("YoY Sales",fontsize=12)
plt.show()
